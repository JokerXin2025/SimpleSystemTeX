\NewDocumentCommand \newconcept {m m} {%

}
\NewDocumentCommand \UserCommand {m d<> +m +d[] +d[]} {
    \Usage{#1-U}(#1)<#2>{#3}
    \IfValueT{#4}{
        \MandatoryArguments{#1-MP}<#2>{#4}
    }

    \IfValueT{#5}{
        \OptionalParameter{#1-OP}<#2>{#5}
    }
    \par\medskip
}
