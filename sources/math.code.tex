% Simple Letter

\clist_map_inline:nn {rm, bf, it, bb, cal, frak, sf, tt, scr} {
    \exp_args:Nc \NewDocumentCommand {UseMathletter#1} {} {
        \int_step_inline:nnn { `a } { `z } {
            \exp_args:Nc \newcommand {#1 \char_generate:nn {####1} {11}} {
                \use:c {math#1} {\char_generate:nn {####1} {11}}
            }
        }
    }
}
